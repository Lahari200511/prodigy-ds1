import pandas as pd
import matplotlib.pyplot as plt

# Load dataset (ensure raw string for file path)
file_path = r"D:\PRODIGY\TASK 1\pop_and_net_migration.csv"
df = pd.read_csv(file_path)

# Display available columns to confirm structure
print("Columns in dataset:", df.columns)

# Filter data for the year 2022
df_2022 = df[df['Year'] == 2022][['Country', 'total_population']].dropna()

# Convert 'total_population' to numeric
df_2022['total_population'] = pd.to_numeric(df_2022['total_population'], errors='coerce')

# Sorting to get top 10 most populous countries
df_top10 = df_2022.sort_values(by='total_population', ascending=False).head(10)

# Ensure data is valid before plotting
if not df_top10.empty:
    # **Bar Chart**
    plt.figure(figsize=(12, 6))
    plt.bar(df_top10['Country'], df_top10['total_population'], color='skyblue')
    plt.xlabel("Country")
    plt.ylabel("Population")
    plt.title("Top 10 Most Populous Countries (2022)")
    plt.xticks(rotation=45)
    plt.grid(axis='y', linestyle='--', alpha=0.7)
    plt.show()

    # **Histogram** (distribution of population)
    plt.figure(figsize=(10, 5))
    plt.hist(df_2022['total_population'], bins=20, color='purple', edgecolor='black', alpha=0.7)
    plt.xlabel("Population")
    plt.ylabel("Frequency")
    plt.title("Distribution of Population Across Countries (2022)")
    plt.grid(axis='y', linestyle='--', alpha=0.7)
    plt.show()
else:
    print("Error: No valid data to plot!")

Columns in dataset: Index(['Country', 'Year', 'total_population', 'net_migration'], dtype='object')
